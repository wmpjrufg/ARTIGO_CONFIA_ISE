def MODULO_ELASTICIDADE_CONCRETO(AGREGADO, F_CK, F_CKJ):
    """
    Esta fun√ß√£o calcula os m√≥dulos de elasticidade do concreto.  

    Entrada:
    AGREGADO    | Tipo de agragado usado no tra√ßo do cimento       |        | string    
                |   'BAS' - Agregado de Basalto                    |        | 
                |   'GRA' - Agregado de Granito                    |        |              
                |   'CAL' - Agregado de Calc√°rio                   |        |
                |   'ARE' - Agregado de Arenito                    |        | 
    F_CK        | Resist√™ncia caracter√≠stica √† compress√£o          | kN/m¬≤  | float   
    F_CKJ       | Resist√™ncia caracter√≠stica √† compress√£o idade J  | kN/m¬≤  | float
    
    Sa√≠da:
    E_CIJ       | M√≥dulo de elasticidade tangente                  | kN/m¬≤  | float
    E_CSJ       | M√≥dulo de elasticidade do secante                | kN/m¬≤  | float   
    """
    # Determina√ß√£o do m√≥dulo tangente E_CI idade T
    if AGREGADO == 'BAS':         
        ALFA_E = 1.2
    elif AGREGADO == 'GRA':         
        ALFA_E = 1.0
    elif AGREGADO == 'CAL':       
        ALFA_E = 0.9
    elif AGREGADO == 'ARE':       
        ALFA_E = 0.7
    F_CK /= 1E3
    if F_CK <= 50:        
        E_CI = ALFA_E * 5600 * np.sqrt(F_CK)
    elif F_CK > 50:   
        E_CI = 21.5 * (10 ** 3) * ALFA_E * (F_CK / 10 + 1.25) ** (1 / 3)
    ALFA_I = 0.8 + 0.2 * F_CK / 80
    if ALFA_I > 1:        
        ALFA_I = 1
    # Determina√ß√£o do m√≥dulo secante E_CS idade T
    E_CS = E_CI * ALFA_I
    if F_CK <= 45 :
        F_CK *= 1E3
        E_CIJ = E_CI * (F_CKJ / F_CK) ** 0.5  
    elif  F_CK > 45 : 
        F_CK *= 1E3
        E_CIJ = E_CI * (F_CKJ / F_CK) ** 0.3  
    E_CSJ = E_CIJ * ALFA_I
    E_CIJ *= 1E3 
    E_CSJ *= 1E3 
    return E_CIJ, E_CSJ

def MOMENTO_RESISTENE_MRD(B_W, D, A_S, F_YK, F_CK, GAMMA_M_ACO, GAMMA_M_CONCRETO):
    """
    Esta fun√ß√£o determina o momento resistente M_RD de uma viga de concreto armado.

    Entrada:
    B_W                | Largura da viga                               | m     | float  
    D                  | Altura √∫til da se√ß√£o                          | m     | float
    A_S                | √Årea de a√ßo necess√°ria na se√ß√£o               | m¬≤    | float
    F_YK               | Tens√£o de escoamento do a√ßo                   | kN/m¬≤ | float
    GAMMA_M_ACO        | Coeficiente parcial de seguran√ßa do a√ßo       |       | float
    GAMMA_M_CONCRETO   | Coeficiente parcial de seguran√ßa do concreto  |       | float

    Sa√≠da:
    M_RD               | Momento resistente da pe√ßa                    | kN.m  | Float 
    """
    # Defini√ß√£o dos par√¢metros que dependem do F_CK
    F_CK /= 1E3
    if F_CK >  50:
        LAMBDA = 0.80 - ((F_CK - 50) / 400)
        ALPHA_C = (1.00 - ((F_CK - 50) / 200)) * 0.85
    else:
        LAMBDA = 0.80
        ALPHA_C = 0.85
    F_CK *= 1E3    
    F_CD = F_CK / GAMMA_M_CONCRETO
    F_YD = F_YK / GAMMA_M_ACO
    # Linha neutra real e momento resistente
    X_III = (A_S * F_YD) / (ALPHA_C * F_CD * B_W * LAMBDA)
    M_RD = A_S * F_YD * (D - 0.50 * LAMBDA * X_III)
    return M_RD

def RESISTENCIA_BIELA_COMPRIMIDA(F_CK, B_W, D, GAMMA_M_CONCRETO):
    """
    Esta fun√ß√£o verifica o valor da resist√™ncia da biela comprimida V_RD2.

    Entrada:
    F_CK               | Resist√™ncia caracter√≠stica √† compress√£o         | kN/m¬≤ | float
    B_W                | Largura da viga                                 | m     | float  
    D                  | Altura √∫til da se√ß√£o                            | m     | float
    GAMMA_M_CONCRETO   | Coeficiente parcial de seguran√ßa do concreto    |       | float
    
    Sa√≠da:
    V_RD2              | Resit√™ncia da biela comprimida                  | kN    | float

def VERIFICA_PILAR(TIPO_PILAR, A_C, RHO, E_S, F_CK, GAMMA_M_CONCRETO):
    """
    Esta fun√ß√£o verifica o valor da carga normal m√°xima em um pilar de concreto armado.
    
    Entrada:
    TIPO_PILAR        | Tipo de pilar da estrutura                    |       | string
                      |    "INTERMEDIARIO"                            |       | 
                      |    "EXTREMIDADE"                              |       | 
                      |    "CANTO"                                    |       | 
    A_C               | √Årea da se√ß√£o de concreto do pilar            | m¬≤    | float
    RHO               | Taxa mec√¢nica de armadura                     | %     | float
    E_S               | M√≥dulo de elasticidade do a√ßo                 | kN/m¬≤ | float
    F_CK              | Resist√™ncia caracter√≠stica √† compress√£o       | kN/m¬≤ | float
    GAMMA_M_CONCRETO  | Coeficiente parcial de seguran√ßa do concreto  |       | float
    
    Sa√≠da:
    N_RD              | Compress√£o resistente do pilar                | m     | float 
    """
    # Defini√ß√£o do coeficiente de convers√£o de flexo-compress√£o para compress√£o
    if TIPO_PILAR == 'CANTO':
        ALPHA = 2.5
    elif TIPO_PILAR == 'EXTREMIDADE':
        ALPHA = 2.2
    elif TIPO_PILAR == 'INTERMEDIARIO':
        ALPHA = 1.8
    # Tens√£o de escoamento do a√ßo a 0.20% de deforma√ß√£o de compress√£o
    SGIMA_SE = 0.2 / 100 * E_S
    # Carga m√°xima resistente
    F_CD = F_CK / GAMMA_M_CONCRETO
    RHO /= 100
    N_RD = (A_C * (0.85 * F_CD + RHO * SGIMA_SE)) / ALPHA
    return N_RD

def MOMENTO_FISSURACAO(GEOMETRIC_FACTOR, F_CK, H, X_I, I_I):
    if GEOMETRIC_FACTOR == 0:
        ALPHA = 1.2
    elif GEOMETRIC_FACTOR == 1:
        ALPHA = 1.3
    elif GEOMETRIC_FACTOR == 2:
        ALPHA = 1.5
    F_CK /= 1E3
    if F_CK <= 50:
          F_CTM = 0.3 * F_CK ** (2/3)
    elif F_CK > 50:
          F_CTM = 2.12 * np.log(1 + 0.11 * F_CK)
    F_CT = F_CTM * 1E3
    Y_T = H - X_I
    M_R = ALPHA * F_CT * (I_I / Y_T)
    return M_R

def GEOMETRIC_PROPERTIES_STATE_I(H, B_F, B_W, H_F, A_SB, ALPHA_MOD, D):
    A_C = (B_F - B_W) * H_F + B_W * H + A_SB * (ALPHA_MOD - 1)
    X_I = ((B_F - B_W) * ((H_F ** 2) / 2) + B_W * ((H ** 2 ) / 2) + A_SB * (ALPHA_MOD - 1) * D) / A_C
    I_I = ((B_F - B_W) * H_F ** 3) / 12 + (B_W * H ** 3) / 12 + (B_F - B_W) * H_F * (X_I - H_F / 2) ** 2 + B_W * H * (X_I - H / 2) ** 2 + A_SB * (ALPHA_MOD - 1) * (X_I - D) ** 2
    return A_C, X_I, I_I

def GEOMETRIC_PROPERTIES_STATE_II(H, B_F, B_W, H_F, A_SB, A_ST, ALPHA_MOD, D, D_L):
    if B_F <= B_W:
        A_1 = B_W / 2
        A_2 = H_F * (B_F - B_W) + (ALPHA_MOD - 1) * A_ST + ALPHA_MOD * A_SB
        A_3 = - D_L * (ALPHA_MOD - 1) * A_ST - D * ALPHA_MOD * A_SB - (H_F ** 2) / 2 * (B_F - B_W)
        X_II = (-A_2 + np.sqrt(A_2 ** 2 - 4 * A_1 * A_3)) / (2 * A_1)
    elif B_F > B_W:
        A_1 = B_F / 2
        A_2 = H_F * (0) + (ALPHA_MOD - 1) * A_ST + ALPHA_MOD * A_SB
        A_3 = -D_L*(ALPHA_MOD - 1) * A_ST - D * ALPHA_MOD * A_SB - (H_F ** 2) / 2 * (0)
        X_II = (- A_2 + (A_2 ** 2 - 4 * A_1 * A_3) ** 0.50) / (2 * A_1)
        if X_II > H_F:
            A_1 = B_W / 2
            A_2 = H_F * (B_F - B_W) + (ALPHA_MOD - 1) * A_ST + ALPHA_MOD * A_SB
            A_3 = - D_L * (ALPHA_MOD - 1) * A_ST - D * ALPHA_MOD * A_SB - (H_F ** 2) / 2 * (B_F - B_W)
            X_II = (- A_2 + np.sqrt(A_2 ** 2 - 4 * A_1 * A_3)) / (2 * A_1)    
    if X_II <= H_F:
        I_II = (B_F * X_II ** 3) / 3 + ALPHA_MOD * A_SB * (X_II - D) ** 2 + (ALPHA_MOD - 1) * A_ST * (X_II - D_L) ** 2
    else:
        I_II = ((B_F - B_W) * H_F ** 3) / 12 + (B_W * X_II **3 ) / 3 + (B_F - B_W) * (X_II - H_F / 2) ** 2 + ALPHA_MOD * A_SB * (X_II - D) ** 2 + (ALPHA_MOD - 1) * A_ST * (X_II - D_L) ** 2
    return X_II, I_II

def BRANSON_INERTIA(M_R, M_D,I_I, I_II):
    M_RMD = (M_R / M_D) ** 3
    I_BRANSON = M_RMD * I_I + (1 - M_RMD) * I_II
    return I_BRANSON
