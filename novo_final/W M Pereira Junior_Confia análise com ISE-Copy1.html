def AREA_SAPATA(N_GK, N_QK, N_SPT, COTA_SAPATA, A_P, B_P):
    """
    Esta fun√ß√£o determina a √°rea de uma sapata em fun√ß√£o do carregamento e da tens√£o adm√≠ssivel do solo.

    Entrada:
    N_GK         | Carga normal permanente caracter√≠stica do pilar             | kN    | float
    N_QK         | Carga normal vari√°vel caracter√≠stica do pilar               | kN    | float
    N_SPT        | Valor do ensaio de sondagem tipo N_SPT para perfil de solo  |       | list
    COTA_SAPATA  | Cota de assentamento da sapata                              | m     | int
    A_P          | Maior dimens√£o do pilar                                     | m     | float
    B_P          | Menor dimens√£o do pilar                                     | m     | float

    Sa√≠da:
    A_SAPATA     | √Årea da sapata                                              | m¬≤    | float
    """
    # Carga de projeto
    N_SK = 1.05 * N_GK + N_QK
    # √Årea da sapata
    B_LISTA = [0.60, 0.65, 0.70, 0.75, 0.80, 0.85, 0.90, 0.95, 1.00, 1.05, 1.10, 1.15, 1.20, 1.25, 1.30, 1.35, 1.40, 1.45]
    CONT = 0
    ERRO = True
    while ERRO:
            B = B_LISTA[CONT]
            SIGMA_SOLOADM, N_SPTMEDIO = TENSAO_ADMISSIVEL(N_SPT, COTA_SAPATA, B)
            A = B - B_P + A_P
            A_SAPATA = A * B
            SIGMA_REAL = N_SK / A_SAPATA
            if SIGMA_SOLOADM >= SIGMA_REAL:
                ERRO = False
            else:
                ERRO = True
            CONT += 1
    return A_SAPATA, A, B, N_SPTMEDIO

def TENSAO_ADMISSIVEL(N_SPT, COTA_SAPATA, B):
    """
    Esta fun√ß√£o determina a tens√£o admiss√≠vel no solo a partir de dados de uma sondagem.

    Entrada:
    N_SPT        | Valor do ensaio de sondagem tipo N_SPT   |       | list
    COTA_SAPATA  | Cota de assentamento da sapata           | m     | int
    B            | Menor dimens√£o da sapata                 | m     | float

    Sa√≠da:
    SIGMA_ADM    | Valor da tens√£o admiss√≠vel m√©dia do solo | kN/m¬≤ | float 
    """
    # Determina√ß√£o da profundidas contidas no bulbo de press√µes
    COTA_BULBO = 2 * B
    COTA_FINALCOMBULBO = int(COTA_SAPATA) + round(COTA_BULBO)
    PERFIL_BULBO = []
    PERFIL_INICIAL = []
    for I_CONT in range(COTA_FINALCOMBULBO + 1):
        PERFIL_BULBO.append(I_CONT) 
    for J_CONT in range(COTA_SAPATA):
        PERFIL_INICIAL.append(J_CONT) 
    PROFUNDIDADES = np.setdiff1d(PERFIL_BULBO, PERFIL_INICIAL)
    N_SPTBULBO = []
    for K_CONT, VALOR_NSPT in enumerate(PROFUNDIDADES):
        N_SPTBULBO.append(N_SPT[VALOR_NSPT])
    # N_SPT m√©dio e tens√£o adm do solo
    N_SPTMEDIO = np.mean(N_SPTBULBO)
    SIGMA_ADM = N_SPTMEDIO / 0.05
    return SIGMA_ADM, N_SPTMEDIO

def MODULO_ELASTICIDADE_CONCRETO(AGREGADO, F_CK, F_CKJ):
    """
    Esta fun√ß√£o calcula os m√≥dulos de elasticidade do concreto.  

    Entrada:
    AGREGADO    | Tipo de agragado usado no tra√ßo do cimento       |        | string

                |   'GRA' - Agregado de Granito                    |        |              
                |   'CAL' - Agregado de Calc√°rio                   |        |
                |   'ARE' - Agregado de Arenito                    |        | 
    F_CK        | Resist√™ncia caracter√≠stica √† compress√£o          | kN/m¬≤  | float   
    F_CKJ       | Resist√™ncia caracter√≠stica √† compress√£o idade J  | kN/m¬≤  | float

        ALFA_E = 1.2
    elif AGREGADO == 'GRA':

        ALFA_E = 0.9
    elif AGREGADO == 'ARE':       
        ALFA_E = 0.7

    if F_CK <= 50:        
        E_CI = ALFA_E * 5600 * np.sqrt(F_CK)
    elif F_CK > 50:   
        E_CI = 21.5 * (10 ** 3) * ALFA_E * (F_CK / 10 + 1.25) ** (1 / 3)
    ALFA_I = 0.8 + 0.2 * F_CK / 80
    if ALFA_I > 1:        
        ALFA_I = 1
    # Determina√ß√£o do m√≥dulo secante E_CS idade T

    if F_CK <= 45 :

        E_CIJ = E_CI * (F_CKJ / F_CK) ** 0.5  
    elif  F_CK > 45 : 
        F_CK *= 1E3
        E_CIJ = E_CI * (F_CKJ / F_CK) ** 0.3  
    E_CSJ = E_CIJ * ALFA_I
    E_CIJ *= 1E3 
    E_CSJ *= 1E3

def MOMENTO_RESISTENE_MRD(B_W, D, A_S, F_YK, F_CK, GAMMA_M_ACO, GAMMA_M_CONCRETO):
    """
    Esta fun√ß√£o determina o momento resistente M_RD de uma viga de concreto armado.

    Entrada:
    B_W                | Largura da viga                               | m     | float  
    D                  | Altura √∫til da se√ß√£o                          | m     | float
    A_S                | √Årea de a√ßo necess√°ria na se√ß√£o               | m¬≤    | float
    F_YK               | Tens√£o de escoamento do a√ßo                   | kN/m¬≤ | float
    GAMMA_M_ACO        | Coeficiente parcial de seguran√ßa do a√ßo       |       | float
    GAMMA_M_CONCRETO   | Coeficiente parcial de seguran√ßa do concreto  |       | float

    Sa√≠da:
    M_RD               | Momento resistente da pe√ßa                    | kN.m  | Float 
    """
    # Defini√ß√£o dos par√¢metros que dependem do F_CK
    F_CK /= 1E3
    if F_CK >  50:
        LAMBDA = 0.80 - ((F_CK - 50) / 400)
        ALPHA_C = (1.00 - ((F_CK - 50) / 200)) * 0.85
    else:
        LAMBDA = 0.80
        ALPHA_C = 0.85
    F_CK *= 1E3    
    F_CD = F_CK / GAMMA_M_CONCRETO
    F_YD = F_YK / GAMMA_M_ACO
    # Linha neutra real e momento resistente
    X_III = (A_S * F_YD) / (ALPHA_C * F_CD * B_W * LAMBDA)
    M_RD = A_S * F_YD * (D - 0.50 * LAMBDA * X_III)
    return M_RD

    if TIPO_PILAR == 'CANTO':

def GEOMETRIC_PROPERTIES_STATE_I(H, B_F, B_W, H_F, A_SB, ALPHA_MOD, D):
    A_C = (B_F - B_W) * H_F + B_W * H + A_SB * (ALPHA_MOD - 1)
    X_I = ((B_F - B_W) * ((H_F ** 2) / 2) + B_W * ((H ** 2 ) / 2) + A_SB * (ALPHA_MOD - 1) * D) / A_C
    I_I = ((B_F - B_W) * H_F ** 3) / 12 + (B_W * H ** 3) / 12 + (B_F - B_W) * H_F * (X_I - H_F / 2) ** 2 + B_W * H * (X_I - H / 2) ** 2 + A_SB * (ALPHA_MOD - 1) * (X_I - D) ** 2
    return A_C, X_I, I_I

def GEOMETRIC_PROPERTIES_STATE_II(H, B_F, B_W, H_F, A_SB, A_ST, ALPHA_MOD, D, D_L):
